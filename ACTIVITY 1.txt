#include <stdio.h>

void twoSum(int nums[], int n, int target) {
    for(int i = 0; i < n; i++) {
        for(int j = i + 1; j < n; j++) {
            if(nums[i] + nums[j] == target) {
                printf("[%d, %d]", i, j);
                return;
            }
        }
    }
}

int main() {
    int nums[] = {2,7,11,15};
    int target = 9;
    int n = sizeof(nums)/sizeof(nums[0]);

    twoSum(nums, n, target);
    return 0;
}

#include <string.h>

int longestSubstring(char s[]) {
    int maxLen = 0;

    for(int i = 0; i < strlen(s); i++) {
        int visited[256] = {0};
        int len = 0;

        for(int j = i; j < strlen(s); j++) {
            if(visited[s[j]] == 1)
                break;

            visited[s[j]] = 1;
            len++;
        }

        if(len > maxLen)
            maxLen = len;
    }

    return maxLen;
}

int main() {
    char s[] = "abcabcbb";
    printf("%d", longestSubstring(s));
    return 0;
}

void productExceptSelf(int nums[], int n) {
    int left[n], right[n], result[n];

    left[0] = 1;
    for(int i = 1; i < n; i++)
        left[i] = left[i-1] * nums[i-1];

    right[n-1] = 1;
    for(int i = n-2; i >= 0; i--)
        right[i] = right[i+1] * nums[i+1];

    for(int i = 0; i < n; i++)
        result[i] = left[i] * right[i];

    for(int i = 0; i < n; i++)
        printf("%d ", result[i]);
}

int main() {
    int nums[] = {1,2,3,4};
    int n = 4;

    productExceptSelf(nums, n);
    return 0;
}

int maxSubArray(int nums[], int n) {
    int maxSum = nums[0];
    int currentSum = nums[0];

    for(int i = 1; i < n; i++) {
        if(currentSum < 0)
            currentSum = nums[i];
        else
            currentSum += nums[i];

        if(currentSum > maxSum)
            maxSum = currentSum;
    }

    return maxSum;
}

int main() {
    int nums[] = {-2,1,-3,4,-1,2,1,-5,4};
    int n = 9;

    printf("%d", maxSubArray(nums, n));
    return 0;
}

int binarySearch(int nums[], int n, int target) {
    int left = 0, right = n - 1;

    while(left <= right) {
        int mid = (left + right) / 2;

        if(nums[mid] == target)
            return mid;
        else if(nums[mid] < target)
            left = mid + 1;
        else
            right = mid - 1;
    }

    return -1;
}

int main() {
    int nums[] = {-1,0,3,5,9,12};
    int target = 9;
    int n = 6;

    printf("%d", binarySearch(nums, n, target));
    return 0;
}
