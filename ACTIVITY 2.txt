#include <stdio.h>
#include <string.h>

void longestPalindrome(char s[]) {
    int start = 0, maxLen = 1;
    int n = strlen(s);

    for(int i = 0; i < n; i++) {
        int l = i, r = i;

        while(l >= 0 && r < n && s[l] == s[r]) {
            if(r - l + 1 > maxLen) {
                start = l;
                maxLen = r - l + 1;
            }
            l--; r++;
        }

        l = i; r = i + 1;
        while(l >= 0 && r < n && s[l] == s[r]) {
            if(r - l + 1 > maxLen) {
                start = l;
                maxLen = r - l + 1;
            }
            l--; r++;
        }
    }

    for(int i = start; i < start + maxLen; i++)
        printf("%c", s[i]);
}

int main() {
    char s[] = "babad";
    longestPalindrome(s);
    return 0;
}
#include <stdio.h>
#include <stdlib.h>

struct Trie {
    struct Trie* children[26];
    int isEnd;
};

struct Trie* createNode() {
    struct Trie* node = (struct Trie*)malloc(sizeof(struct Trie));
    node->isEnd = 0;
    for(int i = 0; i < 26; i++)
        node->children[i] = NULL;
    return node;
}

void insert(struct Trie* root, char* word) {
    for(int i = 0; word[i]; i++) {
        int index = word[i] - 'a';
        if(!root->children[index])
            root->children[index] = createNode();
        root = root->children[index];
    }
    root->isEnd = 1;
}

int search(struct Trie* root, char* word) {
    for(int i = 0; word[i]; i++) {
        int index = word[i] - 'a';
        if(!root->children[index])
            return 0;
        root = root->children[index];
    }
    return root->isEnd;
}

int startsWith(struct Trie* root, char* prefix) {
    for(int i = 0; prefix[i]; i++) {
        int index = prefix[i] - 'a';
        if(!root->children[index])
            return 0;
        root = root->children[index];
    }
    return 1;
}

int subarraySum(int nums[], int n, int k) {
    int count = 0;

    for(int i = 0; i < n; i++) {
        int sum = 0;
        for(int j = i; j < n; j++) {
            sum += nums[j];
            if(sum == k)
                count++;
        }
    }
    return count;
}

int main() {
    int nums[] = {1,1,1};
    int k = 2;
    printf("%d", subarraySum(nums, 3, k));
    return 0;
}
#include <stdio.h>

int search(int nums[], int n, int target) {
    int left = 0, right = n - 1;

    while(left <= right) {
        int mid = (left + right) / 2;

        if(nums[mid] == target)
            return mid;

        if(nums[left] <= nums[mid]) {
            if(target >= nums[left] && target < nums[mid])
                right = mid - 1;
            else
                left = mid + 1;
        } else {
            if(target > nums[mid] && target <= nums[right])
                left = mid + 1;
            else
                right = mid - 1;
        }
    }
    return -1;
}

int main() {
    int nums[] = {4,5,6,7,0,1,2};
    int target = 0;
    printf("%d", search(nums, 7, target));
    return 0;
}
